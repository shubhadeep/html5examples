LocalStorage

Type the following in the developer tools console


    var ls = window.localStorage;

    ls.length;

    ls.setItem("key", "value");

    ls.length;

    ls.getItem("key");
    
    ls.getItem("nonExistentKey");

    ls.removeItem("key");

    ls.length;

    ls.removeItem("nonExistentKey");

    ls.clear();
